    environment:
      MYSQL_ROOT_PASSWORD: admin
      MYSQL_DATABASE: admin

      - "3306:3306"

      - mysql_data:/var/lib/mysql
    networks:

    image: phpmyadmin/phpmyadmin
    container_name: phpmyadmin_container

      - "8484:80"

      - mysql
